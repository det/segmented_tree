499  static void move_next_leaf_count(iterator_entry &entry, size_type count) {
500  auto pointer = entry.leaf.pointer;
501  auto index = entry.leaf.index;
502 
503  // Special case for end iterator.
504  if (pointer == nullptr) {
505  entry.segment.index = entry.segment.length;
506  return;
507  }
508 
509  while (true) {
510  ++index;
511  if (index == pointer->length()) break;
512 
513  auto size = pointer->sizes[index];
514  if (size > count) {
515  entry.leaf.index = index;
516  entry.segment = find_index_segment(
517  cast_segment(pointer->pointers[index]), size, count);
518  return;
519  }
520  count -= size;
521  }
522

524  pointer->parent_index(), count);
525  }
